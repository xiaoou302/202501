import 'package:flutter/material.dart';
import 'package:zenithsprint/features/splash/splash_screen.dart';
import 'package:zenithsprint/features/onboarding/onboarding_screen.dart';
import 'package:zenithsprint/features/goal_selection/goal_selection_screen.dart';
import 'package:zenithsprint/features/baseline_test/baseline_test_screen.dart';
import 'package:zenithsprint/features/initial_report/initial_report_screen.dart';
import 'package:zenithsprint/features/home/home_screen.dart';

class AppRoutes {
  static const String splash = '/';
  static const String onboarding = '/onboarding';
  static const String goalSelection = '/goal-selection';
  static const String baselineTest = '/baseline-test';
  static const String initialReport = '/initial-report';
  static const String home = '/home';

  static Route<dynamic> generateRoute(RouteSettings settings) {
    switch (settings.name) {
      case splash:
        return MaterialPageRoute(builder: (_) => const SplashScreen());
      case onboarding:
        return MaterialPageRoute(builder: (_) => const OnboardingScreen());
      case goalSelection:
        return MaterialPageRoute(builder: (_) => const GoalSelectionScreen());
      case baselineTest:
        return MaterialPageRoute(builder: (_) => const BaselineTestScreen());
      case initialReport:
        return MaterialPageRoute(builder: (_) => const InitialReportScreen());
      case home:
        return MaterialPageRoute(builder: (_) => const HomeScreen());
      default:
        return MaterialPageRoute(builder: (_) => const SplashScreen());
    }
  }
}

import 'package:zenithsprint/features/baseline_test/baseline_test_screen.dart';

class GoalSelectionScreen extends StatefulWidget {
  const GoalSelectionScreen({super.key});

  @override
  _GoalSelectionScreenState createState() => _GoalSelectionScreenState();
}

class _GoalSelectionScreenState extends State<GoalSelectionScreen> {
  String? _selectedGoal;

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Padding(
        padding: const EdgeInsets.all(24.0),
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Text(
              "What's your primary goal?",
              textAlign: TextAlign.center,
              style: Theme.of(context)
                  .textTheme
                  .titleLarge
                  ?.copyWith(fontSize: 28),
            ),
            const SizedBox(height: 48),
            _buildGoalOption(
              'Ace your interviews',
              'Prepare for interviews',
            ),
            const SizedBox(height: 16),
            _buildGoalOption(
              'Boost job performance',
              'Improve work efficiency',
            ),
            const SizedBox(height: 16),
            _buildGoalOption(
              'Prep for standardized tests',
              'Prepare for standardized tests',
            ),
            const SizedBox(height: 48),
            ElevatedButton(
              onPressed: _selectedGoal != null
                  ? () {
                      Navigator.of(context).pushReplacement(
                        MaterialPageRoute(
                          builder: (BuildContext context) =>
                              const BaselineTestScreen(),
                        ),
                      );
                    }
                  : null,
              child: const Text('Continue'),
            ),
          ],
        ),
      ),
    );
  }

  Widget _buildGoalOption(String title, String subtitle) {
    final bool isSelected = _selectedGoal == title;
    return GestureDetector(
      onTap: () {
        setState(() {
          _selectedGoal = title;
        });
      },
      child: Container(
        padding: const EdgeInsets.all(16),
        decoration: BoxDecoration(
          color: isSelected ? Colors.white24 : Colors.white10,
          borderRadius: BorderRadius.circular(12),
          border: Border.all(
            color: isSelected ? Colors.white : Colors.white24,
            width: 2,
          ),
        ),
        child: Row(
          children: [
            Expanded(
              child: Column(
                crossAxisAlignment: CrossAxisAlignment.start,
                children: [
                  Text(
                    title,
                    style: const TextStyle(
                      color: Colors.white,
                      fontSize: 18,
                      fontWeight: FontWeight.bold,
                    ),
                  ),
                  const SizedBox(height: 4),
                  Text(
                    subtitle,
                    style: const TextStyle(
                      color: Colors.white70,
                      fontSize: 14,
                    ),
                  ),
                ],
              ),
            ),
            if (isSelected)
              const Icon(
                Icons.check_circle,
                color: Colors.white,
              ),
          ],
        ),
      ),
    );
  }
}

Zenith Sprint - Flutter App 架构与需求文档
1. 项目概述
Zenith Sprint 是一款专注于提升用户在压力场景下心算能力的移动应用。它通过模拟高压职业场景（如咨询面试、项目报价等）中的计算任务，帮助用户将计算能力从有意识的负担转变为无意识的直觉。应用的核心是提供快速、专注、高强度的自适应训练，并通过专业级的认知诊断报告，让用户清晰地看到自己的成长。

2. 核心功能
用户引导流程 (Onboarding)

价值展示: 通过几个引导页向新用户展示应用的核心价值。

目标设定: 允许用户选择他们的主要训练目标（如备战面试、提升工作效率等）。

基准测试: 通过一个简短的测试来评估用户的初始心算水平。

初始报告: 根据测试结果生成一份初始的认知能力图谱。

核心训练模块 (Training)

自适应冲刺 (Adaptive Sprint): App 的主要训练模式，提供可自定义时长的、个性化的心算题组。

压力舱 (The Pressure Chamber): 模拟真实职业场景的挑战模式（原型阶段为入口展示）。

专项训练: 按特定技能（如百分比、估算）进行分类的训练入口。

训练界面: 极简的答题界面，包含题目展示区和为数字输入优化的自定义键盘。

结果反馈: 训练结束后，展示本次训练的准确率、答题数等核心数据，并有几率解锁“认知技巧卡片”。

认知仪表盘 (Insights)

认知档案 (Cognitive Profile): 通过雷达图等多维度可视化图表，展示用户的综合心算能力。

成长趋势: 通过趋势图展示用户在特定指标（如压力下准确率）上的历史进步。

设置与信息 (Settings)

提供“关于我们”、“用户协议”、“隐私协议”、“使用帮助”和“反馈和建议”等静态信息页面的入口。

3. App 架构设计
为了保持低耦合度和清晰的结构，我们采用 按功能模块组织 (Feature-Driven) 的目录结构。

状态管理: 初期完全使用 StatefulWidget 和 setState 进行状态管理。每个页面或复杂组件负责管理自身的 UI 状态，避免跨页面或跨模块的复杂状态依赖。

数据存储: 使用 shared_preferences 库进行轻量级的本地数据持久化。主要用于存储用户的设置、认知档案数据和训练历史。我们会创建一个服务类来封装其读写操作，以便将数据模型（Dart Class）转换为 JSON 字符串进行存储。

导航: 使用 Flutter 内置的 Navigator 进行页面路由管理。

UI 设计:

主题 (Theme): 在 main.dart 中定义全局的 ThemeData，统一管理应用的颜色（深邃海军蓝、活力翡翠绿等）、字体（使用系统默认字体）和通用组件样式。

响应式布局: 使用 MediaQuery, LayoutBuilder, Flex 等组件确保 UI 在不同尺寸的设备上表现良好。
